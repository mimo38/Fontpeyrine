\couplet{
	Ave Maris Stella,\\*
	Dei Mater alma,\\*
	Atque semper virgo,\\*
	Felix cæli porta.
}

\couplet{
	Sumens illud Ave\\*
	Gabriélis ore,\\
	Funda nos in pace,\\*
	Mutans Evæ nomen.
}

\couplet{
	Solve vincla reis,\\*
	Profer lumen cæcis ;\\
	Mala nostra pelle,\\*
	Bona cuncta posce.
}

\couplet{
	Montra t\emph{e} esse matrem,\\*
	Sumat per te preces\\
	Qui, pro nobis natus,\\*
	Tulit esse tuus.
}

\couplet{
	Virgo singuláris,\\*
	Inter omnes mitis,\\
	Nos culpis solútos,\\*
	Mites fac et castos.
}

\couplet{
	Vitam præsta puram,\\*
	Iter para tutum,\\
	Ut vidéntes Iesum,\\*
	Semper collætémur.
}

\couplet{
	Sit laus Deo Patri,\\*
	Summo Christo decus ;\\*
	Spirítui Sancto,\\*
	Tribus honor unus.
	Amen.
}
