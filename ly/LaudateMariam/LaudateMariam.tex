\couplet{
	Ô Vierge Marie,\\*
	Entends près de Dieu\\
	Ton peuple qui prie :\\*
	Exauce ses vœux.
}

\refrain{
	\bis{%
		Laudáte, laudáte,\\*
		Laudáte Maríam !%
	}
}

\couplet{
	Unis aux saints Anges\\*
	Devant ton autel,\\
	Reçois nos louanges,\\*
	Ô Reine du Ciel.
}

\couplet{
	Ton âme très pure\\*
	Est sainte en naissant.\\
	De toute souillure\\*
	Le Ciel la défend.
}

\couplet{
	Le Verbe adorable\\*
	Fait homme pour nous\\
	De toi, Vierge aimable,\\*
	Veut naître humble et doux.
}

\couplet{
	Et Dieu te couronne\\*
	Devant les élus,\\
	Ton trône rayonne\\*
	Auprès de Jésus.
}

\couplet{
	Ô Reine bénie,\\*
	Qu’au sein des splendeurs\\
	Après cette vie\\*
	Te chantent nos cœurs.
}
